路由到页面的代码

```text
import { TitleBar } from '../../webContainer/pages/TitleBar'
import App from '@system.app'

@Entry
@Component
export struct ElecCardPage {

  build() {
    // 需要 NavDestination包裹
    NavDestination(){
      Column(){
        TitleBar({
          title: "电子名片",
          bottomLineVisibility: false
        })
      }
    }
    .hideTitleBar(true)
    .width('100%').height('100%')
  }
}

// 需要提供路由表内指定的方法
@Builder
export function getElecCardPage() {
  ElecCardPage();
}
```
